from flask import Flask, render_template, request app = Flask(__name__) # Lista para guardar los registros registros = [] @app.route("/", methods=["GET", "POST"]) def index(): if request.method == "POST": # Obtener datos del formulario nombre = request.form.get("nombre") correo = request.form.get("correo") ip = request.remote_addr # IP del visitante # Guardar en registros registros.append({"nombre": nombre, "correo": correo, "ip": ip}) mensaje = f"Registro recibido: {nombre}, {correo}, IP: {ip}" return render_template("index.html", mensaje=mensaje, registros=registros) return render_template("index.html", registros=registros) if __name__ == "__main__": app.run(debug=True)
